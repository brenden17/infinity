~~~

* svn

~~~
$sudo apt-get install subversion
~~~

* mercurial
 
~~~
$sudo apt-get install mercurial
~~~

* git

~~~
$sudo apt-get install git

* virtualenv

~~~    
$sudo apt-get install python-setuptools

~~~
